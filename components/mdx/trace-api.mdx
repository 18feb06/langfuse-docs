```bash
# Create trace
curl --location 'http://localhost:3000/api/traces' \
--header 'Content-Type: application/json' \
--header 'Accept: application/json' \
--data '{
  "name": "<string>",
  "attributes": {},
  "status": "<string>",
  "statusMessage": "<string>"
}'

# Create span
curl --location 'http://localhost:3000/api/spans' \
--header 'Content-Type: application/json' \
--header 'Accept: application/json' \
--data '{
  "traceId": "<string>",
  "name": "<string>",
  "startTime": "<dateTime>",
  "attributes": {},
  "parentObservationId": "<string>"
}'

# Create event
curl --location 'http://localhost:3000/api/events' \
--header 'Content-Type: application/json' \
--header 'Accept: application/json' \
--data '{
  "traceId": "<string>",
  "name": "<string>",
  "startTime": "<dateTime>",
  "attributes": {},
  "parentObservationId": "<string>"
}'

# Log a metric
curl --location 'http://localhost:3000/api/metrics' \
--header 'Content-Type: application/json' \
--header 'Accept: application/json' \
--data '{
  "traceId": "<string>",
  "name": "<string>",
  "value": "<integer>",
  "observationId": "<string>"
}'
```
